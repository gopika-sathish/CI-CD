pipeline {
    agent any

    stages {
        stage('Clone') {
            steps {
                git 'https://github.com/your-username/resume-project.git'
            }
        }

        stage('Build Docker Image') {
            steps {
                sh 'docker build -t resume-app .'
            }
        }

        stage('Run Container') {
            steps {
                sh 'docker run -d -p 8081:80 resume-app'
            }
        }
    }
}

# ✅ 2. App.java (Shopping Cart Logic)

📄 `src/main/java/com/example/App.java`

```java
package com.example;

import java.util.ArrayList;
import java.util.List;

public class App {

    private List<Integer> cart = new ArrayList<>();

    // Add item price
    public void addItem(int price) {
        cart.add(price);
    }

    // Remove item price
    public void removeItem(int price) {
        cart.remove(Integer.valueOf(price));
    }

    // Calculate total
    public int getTotal() {
        int total = 0;
        for (int price : cart) {
            total += price;
        }
        return total;
    }

    // Main method (for running)
    public static void main(String[] args) {
        App cart = new App();
        cart.addItem(100);
        cart.addItem(200);
        System.out.println("Total: " + cart.getTotal());
    }
}
```

        App cart = new App();
        cart.addItem(100);
        cart.addItem(200);
        assertEquals(300, cart.getTotal());
    }

    @Test
    public void testRemoveItem() {
        App cart = new App();
        cart.addItem(100);
        cart.addItem(200);
        cart.removeItem(100);
        assertEquals(200, cart.getTotal());
    }
}
